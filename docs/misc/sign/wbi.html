function encWbi(params, img_key, sub_key) {

        curr_time = Math.round(Date.now() / 1000),
        chr_filter = /[!'\(\)*]/g
    let query = []
    params = Object.assign(params, {wts: curr_time})    // æ·»åŠ  wts å­—æ®µ
    // æŒ‰ç…§ key é‡æ’å‚æ•°
